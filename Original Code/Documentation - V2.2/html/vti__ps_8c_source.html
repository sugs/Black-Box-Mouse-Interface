00481

00484                 if (h0[i] > hnoll) break;

00493         // Apply correction factor to pressure table

00502 // @fn          conv_pa_to_meter
00503 // @brief       Convert pressure (Pa) to altitude (m) using a conversion table
00504 //                              Implemented straight from VTI reference code.

00526                 if (p[i] < fl_p_meas) break;

00540                 hnoll = (float)(fl_p_meas - p_low)/(p[i] - p_low)*((float)(h0[i] - h0[i-1])) + h0[i-1];
