00609        * frame. The AP is only MAX_HOPS_FROM_AP hops away from us.

00641                         ((uint32_t)(MRFI_RandomByte())<<16) | \
00642                         ((uint32_t)(MRFI_RandomByte())<<24);
00643 
00644     nwk_putNumObjectIntoMsg((void *)&pCInfo->connTxCTR, (void *)&msg[LR_CTR_OS], 4);
00645 #endif
00646     if (pOutFrame = nwk_buildFrame(SMPL_PORT_LINK, msg, sizeof(msg), MAX_HOPS-(GET_FROM_FRAME(MRFI_P_PAYLOAD(frame),F_HOP_COUNT))))
00647     {
00648       /* destination address is the source adddress of the received frame. */
00649       memcpy(MRFI_P_DST_ADDR(&pOutFrame->mrfiPkt), MRFI_P_SRC_ADDR(frame), NET_ADDR_SIZE);
00650 #if defined(SMPL_SECURE)
00651       nwk_setSecureFrame(&pOutFrame->mrfiPkt, sizeof(msg), 0);
00652 #endif
00653       if (SMPL_SUCCESS != nwk_sendFrame(pOutFrame, MRFI_TX_TYPE_FORCED))
00654       {
00655         /* better release the connection structure */
00656         nwk_freeConnection(pCInfo);
00657       }
00658     }
00659     else
00660     {
00661       /* better release the connection structure */
00662       nwk_freeConnection(pCInfo);
00663     }
00664   }
00665   /* we're done with the packet */
00666   return SENT_REPLY;
00667 #else
00668   return SENT_NO_REPLY;
00669 #endif  /* NUM_CONNECTIONS */

00675  * @brief       Process Link frame. Just save the frame for the Link app if it

00815 
00816   if (LINK_LEGACY_MSG_LENGTH == (MRFI_GET_PAYLOAD_LEN(frame) - F_APP_PAYLOAD_OS))

00819     *(MRFI_P_PAYLOAD(frame)+F_APP_PAYLOAD_OS) = LINK_REQ_LINK;

00822   switch (*(MRFI_P_PAYLOAD(frame)+F_APP_PAYLOAD_OS))

00824     case LINK_REQ_LINK:
00825       isReplySent = smpl_send_link_reply(frame);
00826 #if !defined(END_DEVICE)

00843 #endif
00844
