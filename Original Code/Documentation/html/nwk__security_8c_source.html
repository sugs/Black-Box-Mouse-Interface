00049 /*                   *** GENERAL SECURITY OUTLINE ***
00050  *
00051  * We are using XTEA (eXtended Tiny Encryption Algorithm) with a fixed
00052  * number of rounds (32). We have removed the parameters from the API
00053  * we harvested from the public domain.
00054  *
00055  * We are using a CTR-like mode. We use the 64-bit block cipher function of the
00056  * XTEA code to encipher a concatenation of the 32-bit initialization vector and
00057  * a 32-bit counter that increments each block. We encrypt using a fixed 128-bit
00058  * key. The resulting 64-bit output is XOR'ed with the message. If the message is
00059  * longer than 64 bits we encipher the next block (incrementing the counter) and
00060  * continue until the message is exhausted. If the last cipher block is longer
00061  * than the message we simply discard the remaining cipher block.
00062  */

00191  * @fn          msg_encipher
00192  *
00193  * @brief       Encipher a message using the XTEA algorithm and the modified
00194  *              CTR mode method.
00195  *
00196  * input parameters
00197  * @param   msg      - pointer to message to encipher
00198  * @param   len      - length of message
00199  * @param   cntStart - pointer to the counter used in the cipher block.
00200  *
00201  * output parameters
00202  * @param   cntStart - counter is updated during encryption.
00203  *
00204  * @return      void
00205  */

00207 static void msg_encipher(uint8_t *msg, uint8_t len, uint32_t *cntStart)
00208 {
00209   uint8_t  i, idx, done;
00210   uint8_t *mptr = (uint8_t *)&sMsg[0];
00211   uint32_t ctr;
00212 
00213   if ((NULL == msg) || !len)
00214   {
00215     return;
00216   }
00217 
00218   /* set local counter from input */
00219   ctr = *cntStart;
00220 
00221   idx  = 0;
00222   done = 0;
00223   do
00224   {
00225     /* Set block to be enciphered. 1st 32 bits are the IV. The second
00226      * 32 bits are the current CTR value.
00227      */
00228     sMsg[0] = sIV;
00229     sMsg[1] = ctr;
00230     /* encrypt */
00231     xtea_encipher();
00232     /* increment counter for next time. */
00233     ctr++;
00234     /* XOR ciphered block with message to be sent. Only operate
00235      * up to and including the last message byte which may not
00236      * be on a cipher block boundary (64 bits == 8 bytes).
00237      */
00238     for (i=0; i<sizeof(sMsg) && idx<len; ++i, ++idx)
00239     {
00240       msg[idx] ^= mptr[i];
00241     }
00242 
00243     if (idx >= len)
00244     {
00245       /* we're done */
00246       done = 1;
00247     }
00248   } while (!done);
00249 
00250   /* return counter value start for next time */
00251   *cntStart = ctr;
00252 
00253   return;
00254 }

00279 /******************************************************************************
00280  * @fn          xtea_encipher
00281  *
00282  * @brief       XTEA encipher algorithm. Calling arguments removed from public
00283  *              domain code and static-scope values used instead.
00284  *
00285  * input parameters
00286  *
00287  * output parameters
00288  *
00289  * @return      void
00290  */
00291 void xtea_encipher(void)
00292 {
00293   uint32_t v0=sMsg[0], v1=sMsg[1];
00294   uint16_t i;
00295   uint32_t sum=0, delta=0x9E3779B9;
00296 
00297   for(i=0; i<NUM_ROUNDS; i++)
00298   {
00299     v0  += (((v1 << 4) ^ (v1 >> 5)) + v1) ^ (sum + sKey.keyL[sum & 3]);
00300     sum += delta;
00301     v1  += (((v0 << 4) ^ (v0 >> 5)) + v0) ^ (sum + sKey.keyL[(sum>>11) & 3]);
00302   }
00303 
00304   sMsg[0]=v0;
00305   sMsg[1]=v1;
00306 }

00308 /******************************************************************************
00309  * @fn          nwk_setSecureFrame
00310  *
00311  * @brief       Called from NWK to secure a frame.
00312  *
00313  * input parameters
00314  * @param   frame   - pointer to frame to secure
00315  * @param   msglen  - length of message
00316  * @param   ctr     - pointer to the counter used in the cipher block. This will
00317  *                    be NULL if a network application is sending a frame. Since
00318  *                    these are not connection-based there is no counter sync
00319  *                    issue but we still need a counter value. A random value
00320  *                    is used.
00321  *
00322  * output parameters
00323  * @param   cntStart - counter is updated during encryption.
00324  *
00325  * @return      void
00326  */
00327 void nwk_setSecureFrame(mrfiPacket_t *frame, uint8_t msglen, uint32_t *ctr)
00328 {
00329   uint32_t locCnt;
00330 
00331   /* If an encrypted frame is to be sent to a non-connection based port use a
00332    * random number as the lsb counter value. In this case only the lsb is used
00333    * for a counter value during decryption. Not as secure but there are still
00334    * the 32 bits in the IV.
00335    */
00336   locCnt = ctr ? *ctr : MRFI_RandomByte();
00337 
00338   /* place counter value into frame */
00339   PUT_INTO_FRAME(MRFI_P_PAYLOAD(frame), F_SEC_CTR_OS, (uint8_t)(locCnt & 0xFF));
00340 
00341   /* Put MAC value in */
00342   nwk_putNumObjectIntoMsg((void *)&sMAC, (void *)(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS), sizeof(secMAC_t));
00343 
00344   /* Put FCS value in */
00345   {
00346     secFCS_t fcs = calcFCS(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS, msglen+sizeof(secMAC_t));
00347 
00348     nwk_putNumObjectIntoMsg((void *)&fcs, (void *)(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS), sizeof(secFCS_t));
00349   }
00350 
00351   /* Encrypt frame */
00352   msg_encipher(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS, msglen+sizeof(secMAC_t)+sizeof(secFCS_t), &locCnt);
00353 
00354   /* Set the Encryption bit */
00355   PUT_INTO_FRAME(MRFI_P_PAYLOAD(frame), F_ENCRYPT_OS, F_ENCRYPT_OS_MSK);
00356 
00357   /* Update the counter if it was a "real" counter. */
00358   if (ctr)
00359   {
00360     *ctr = locCnt;
00361   }
00362 
00363   return;
00364 }

00395 /******************************************************************************
00396  * @fn          nwk_getSecureFrame
00397  *
00398  * @brief       Called from NWK to get a secure a frame and decrypt.
00399  *
00400  * input parameters
00401  * @param   frame    - pointer to frame containing encrypted message
00402  * @param   msglen   - length of message
00403  * @param   ctr      - pointer to the counter used in the cipher block. This will
00404  *                     be NULL if a network applicaiton is getting a frame. Since
00405  *                     these are not connection-nbased there is no counter sync
00406  *                     issue but we still need a counter value.
00407  *
00408  * output parameters
00409  * @param   cntStart - counter is updated during decryption. If decryption fails
00410  *                     this value is not changed.
00411  *
00412  * @return      Returns non-zero if frame decryption is valid, otherwise returns 0.
00413  */
00414 uint8_t nwk_getSecureFrame(mrfiPacket_t *frame, uint8_t msglen, uint32_t *ctr)
00415 {
00416   uint8_t  rc = 1;
00417   uint8_t  done = 0;
00418   uint8_t  cntHint = GET_FROM_FRAME(MRFI_P_PAYLOAD(frame), F_SEC_CTR_OS);
00419   uint32_t locCnt, frameCnt;
00420 
00421   /* Construct proposed CTR values */
00422 
00423   /* Just like encryption, we may be talking to a non-connection based
00424    * peer in which case the counter value is represented by the lsb byte
00425    * conveyed in the frame.
00426    */
00427   locCnt = ctr ? *ctr : cntHint;
00428 
00429   frameCnt = (locCnt & 0xFFFFFF00) + cntHint;
00430 
00431   do
00432   {
00433     /* See if counters match */
00434     if (locCnt == frameCnt)
00435     {
00436       /* When the counters appear to match is the only time we actually decipher
00437        * the message. It is the only time we can do so since out-of-sync lsb counter
00438        * values guarantees that something is wrong somewhere. Decryption is successful
00439        * only if the MAC and FCS values match. The message is left as-is after the
00440        * decipher attempt. Either it appears valid or is doesn't and is discarded.
00441        * There is no recovery attempt if the counters match but the MAC or FCS do
00442        * not. It is considered a rogue message.
00443        */
00444       msg_decipher(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS, msglen-1, &locCnt);
00445 
00446       /* Get MAC and make sure it matches. A failure can occur if a replayed frame happens
00447        * to have the correct counter sync value but was encoded with the wrong complete
00448        * counter value. Otherwise the MAC values must match when the counter values are equal.
00449        */
00450       {
00451         secMAC_t mac;
00452 
00453         nwk_getNumObjectFromMsg((void *)(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS), (void *)&mac, sizeof(secMAC_t));
00454         if (mac != sMAC)
00455         {
00456           rc = 0;
00457         }
00458       }
00459 
00460       /* FCS check... */
00461       {
00462         secFCS_t fcs;
00463 
00464         nwk_getNumObjectFromMsg((void *)(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS), (void *)&fcs, sizeof(secFCS_t));
00465         if (fcs != calcFCS(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS, msglen-1-sizeof(secMAC_t)))
00466         {
00467           rc = 0;
00468         }
00469       }
00470 
00471       /* we're done. */
00472       done = 1;
00473     }
00474     else
00475     {
00476       /* Uh oh. Counters don't match. Try and resync. We need to distinguish among
00477        * missed frames, duplicates and rogues plus account for counter wrap.
00478        */
00479       if (frameCnt > locCnt)
00480       {
00481         /* frameCnt is bigger. Second part of test below takes care of
00482          * the unlikely case of a complete counter wrap (msb's all 0) in
00483          * which case the test will incorrectly fail when the count is
00484          * actually within the (wrapped) window. #ifdef'ed to avoid compiler
00485          * warning in case user sets CNT_WINDOW to 0 (pointless comparison of
00486          * unsigned value).
00487          */
00488         if (((frameCnt-CTR_WINDOW) <= locCnt)
00489 #if CTR_WINDOW > 0
00490             || (frameCnt < CTR_WINDOW)
00491 #endif
00492            )
00493         {
00494           /* Value within window. We probably missed something. Adjust and decipher.
00495            * If locCnt is less because it wrapped and frameCnt didn't it means that
00496            * it's a duplicate or late frame. In that case the following will lead to
00497            * a decryption that fails sanity checks which is OK because the frame will
00498            * be correctly rejected.
00499            */
00500           locCnt = frameCnt;
00501         }
00502         else
00503         {
00504           /* It's either a rogue or a really old duplicate packet. In either case
00505            * we dismiss the frame.
00506            */
00507           rc   = 0;
00508           done = 1;
00509         }
00510       }
00511       else
00512       {
00513         /* locCnt is bigger. The only way the frame can be valid is if the
00514          * counter wrapped causing frameCnt to appear to be smaller. Wrap the
00515          * counter and decrypt. If the frame isn't valid, i.e., it's late,
00516          * a duplicate, or a rogue, the decryption will fail sanity checks and
00517          * the frame will be correctly rejected. The following arithmetic works
00518          * correctly without a special test for the complete counter wrap case.
00519          */
00520         frameCnt += 0x100;   /* wrap the hint-based counter */
00521         if (((frameCnt-CTR_WINDOW) <= locCnt))
00522         {
00523           /* An lsb wrap but still within window. We probably missed something.
00524            * Adjust (with wrap) and decrypt.
00525            */
00526           locCnt = frameCnt;
00527         }
00528         else
00529         {
00530           /* rogue frame */
00531           rc   = 0;
00532           done = 1;
00533         }
00534       }
00535     }
00536   } while (!done);
00537 
00538   if (ctr && rc)
00539   {
00540     /* Only update the counter if the count was a "real" one and the
00541      * decryption succeeded.
00542      */
00543     *ctr = locCnt;
00544   }
00545 
00546   return rc;
00547 }
