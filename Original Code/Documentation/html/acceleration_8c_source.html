00120 // *************************************************************************************************
00121 // @fn          convert_acceleration_value_to_mgrav
00122 // @brief       Converts measured value to mgrav units
00123 // @param       u8 value    g data from sensor 
00124 // @return      u16         Acceleration (mgrav)
00125 // *************************************************************************************************
00126 u16 convert_acceleration_value_to_mgrav(u8 value)
00127 {
00128     u16 result;
00129     u8 i;
00130     
00131     if (!acceleration_value_is_positive(value))
00132     {
00133         // Convert 2's complement negative number to positive number
00134         value = ~value;
00135         value += 1;
00136     }
00137     
00138     result = 0;
00139     for (i=0; i<7; i++)
00140     {
00141         result += ((value & (BIT(i)))>>i) * mgrav_per_bit[i];
00142     }
00143     
00144     return (result);
00145 }
