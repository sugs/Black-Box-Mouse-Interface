00275  * This approach requires that the disptach thread guarantee that it will

00288

00290 
00291   nwk_replayFrame(fiptr);

00339 {
00340 #ifdef ACCESS_POINT
00341   uint8_t      msg[FREQ_REQ_PING_FRAME_SIZE];

00343 
00344   /* original request with reply bit on */
00345   msg[FB_APP_INFO_OS] = *(MRFI_P_PAYLOAD(frame)+F_APP_PAYLOAD_OS) | NWK_APP_REPLY_BIT;
00346   msg[FB_TID_OS]      = *(MRFI_P_PAYLOAD(frame)+F_APP_PAYLOAD_OS+FB_TID_OS);

00352     /* must use transaction ID of source frame */
00353     PUT_INTO_FRAME(MRFI_P_PAYLOAD(&pOutFrame->mrfiPkt), F_TRACTID_OS, (GET_FROM_FRAME(MRFI_P_PAYLOAD(frame), F_TRACTID_OS)));
00354 #ifdef SMPL_SECURE
00355     nwk_setSecureFrame(&pOutFrame->mrfiPkt, sizeof(msg), 0);
00356 #endif  /* SMPL_SECURE */
00357     nwk_sendFrame(pOutFrame, MRFI_TX_TYPE_FORCED);

00377 {
00378   fhStatus_t rc = FHS_RELEASE;
00379   uint8_t    replyType;
00380 
00381   /* Make sure this is a reply and see if we sent this. Validate the

00385   {

00387      * returning a 1 so it can be received by the client app.
00388      */
00389     MRFI_PostKillSem();
00390     rc = FHS_KEEP;
00391   }
00392 #if !defined( END_DEVICE )
00393   else if (SMPL_A_REPLY == replyType)
00394   {
00395     /* no match. if i'm not an ED this is a reply that should be passed on. */
00396     rc = FHS_REPLAY;
00397   }
00398 #endif  /* !END_DEVICE */
00399   else if (SMPL_NOT_REPLY == replyType)
00400   {
00401     rc = handle_freq_cmd(frame);
00402   }
00403

00407 /******************************************************************************
00408  * @fn          nwk_scanForChannels
00409  *
00410  * @brief       Scan for channels by sending a ping frame on each channel in the

00413  * input parameters

00415  *                       ping replies were received.

00417  * output parameters

00421  */
00422 uint8_t nwk_scanForChannels(freqEntry_t *channels)
00423 {

00425   addr_t      *apAddr, retAddr;
00426   uint8_t      radioState = MRFI_GetRadioState();
00427   freqEntry_t  chan;
00428   freqEntry_t  curChan;
00429

00431   {
00432     ioctlRawSend_t    send;
00433     ioctlRawReceive_t recv;
00434   } ioctl_info;
00435 
00436   nwk_getChannel(&curChan);
00437 
00438   /* send to AP. If we don't know AP address, broadcast. */
00439   apAddr = (addr_t *)nwk_getAPAddress();

00441   {
00442     apAddr = (addr_t *)nwk_getBCastAddress();
00443     notBcast = 0;

00445 
00446   for (i=0; i<NWK_FREQ_TBL_SIZE; ++i)
00447   {
00448     chan.logicalChan = i;
00449 
00450     nwk_setChannel(&chan);
00451 
00452     ioctl_info.send.addr = apAddr;
00453     ioctl_info.send.msg  = msg;
00454     ioctl_info.send.len  = sizeof(msg);
00455     ioctl_info.send.port = SMPL_PORT_FREQ;
00456 
00457     msg[FB_APP_INFO_OS] = FREQ_REQ_PING;
00458     msg[FB_TID_OS]      = sTid;
00459 
00460     SMPL_Ioctl(IOCTL_OBJ_RAW_IO, IOCTL_ACT_WRITE, &ioctl_info.send);

00462     ioctl_info.recv.port = SMPL_PORT_FREQ;

00476       {
