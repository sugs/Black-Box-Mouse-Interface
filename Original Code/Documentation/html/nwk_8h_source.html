00101 /****************************************************************************************
00102  *                         Application Payload TID support
00103  *
00104  * Sometimes the receiving application uses a payload TID to determine if either it
00105  * missed frames (received TID > (last-seen TID+1)) or is seeing a duplicate (received
00106  * TID <= last-seen TID). Typically the TID simply increments for each successive frame.
00107  * But when the count wraps there is a problem. The received TID should always be one
00108  * more than the last TID. If it's equal, it's a duplicate. If it's less it's late. If
00109  * it's one more it's the one we expect. If it's more than 1 more then we missed frames.
00110  * Simple increments work for the wrap arithmetically. If the received TID is 0 and the last
00111  * seen ID is the biggest number in the world -- 0xFF... depending on type we can detect
00112  * the wrap. But if the receiver misses the 0 TID value for any reason or the biggest
00113  * number in the world TID is missed then susbequent testing for missed or duplicate
00114  * frames can get confused. We resolve this by allowing some leeway in the wrap testing.
00115  * this testing is assisted by the following macros. Setting TID_VALID_WINDOW to 0
00116  * will enforce a no leniency policy. In this case you'd better not miss either the
00117  * biggest number or the 0. The CHECK_TID_WRAP macro is only needed if the received
00118  * TID is less than the last-seen TID. The DUP_TID_AFTER_WRAP macro is only needed if the
00119  * received TID is greater than 1 more than the last-seen TID.
00120  ***************************************************************************************/
00121 #define MAX_APT           ((appPTid_t)~0)    /* max value of application payload TID type */
00122 #define TID_VALID_WINDOW  2                  /* window around max and 0 */
00123 
00124 #define CHECK_TID_WRAP(lastTID, newTID)   ((lastTID >= (MAX_APT - TID_VALID_WINDOW)) &&  \
00125                                            (newTID <= TID_VALID_WINDOW)                  \
00126                                           )
00127 #define DUP_TID_AFTER_WRAP(lastTID, newTID)     ((newTID >= (MAX_APT - TID_VALID_WINDOW)) &&  \
00128                                                  (lastTID <= TID_VALID_WINDOW)                  \
00129                                                 )
