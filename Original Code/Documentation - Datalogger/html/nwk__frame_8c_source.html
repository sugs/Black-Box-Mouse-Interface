00165  *
00166  * @brief       Builds an output frame for the port and message enclosed.
00167  *              This routine prepends the frame header and populates the
00168  *              frame in the output queue.
00169  *
00170  * input parameters
00171  * @param   port    - port from application
00172  * @param   msg     - pointer to message from app to be sent
00173  * @param   len     - length of enclosed message
00174  * @param   hops    - number of hops allowed. this is less than MAX_HOPS
00175  *                    whenever the frame is being sent to the AP. this is to
00176  *                    help mitigate the (short) broadcast storms
00177  *
00178  * output parameters
00179  *
00180  * @return   pointer to frameInfo_t structure created. NULL if there is
00181  *           no room in output queue.
00182  */
00183 frameInfo_t *nwk_buildFrame(uint8_t port, uint8_t *msg, uint8_t len, uint8_t hops)
00184 {
00185   frameInfo_t  *fInfoPtr;
00186 
00187   if (!(fInfoPtr=nwk_QfindSlot(OUTQ)))
00188   {
00189     return (frameInfo_t *)NULL;
00190   }
00191 
00192   MRFI_SET_PAYLOAD_LEN(&fInfoPtr->mrfiPkt, len+F_APP_PAYLOAD_OS);
00193 
00194   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_ENCRYPT_OS, 0);
00195   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_PORT_OS, port);
00196   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_TRACTID_OS, sTRACTID);
00197   while (!(++sTRACTID)) ;  /* transaction ID can't be 0 */
00198   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_RX_TYPE, sMyRxType);
00199   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_HOP_COUNT, hops);
00200 
00201   /* reset ack-relevant bits */
00202   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_ACK_REQ, 0);
00203   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_ACK_RPLY, 0);
00204 
00205   /* reset forwarding bit */
00206   PUT_INTO_FRAME(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt), F_FWD_FRAME, 0);
00207 
00208   memcpy(MRFI_P_PAYLOAD(&fInfoPtr->mrfiPkt)+F_APP_PAYLOAD_OS, msg, len);
00209   memcpy(MRFI_P_SRC_ADDR(&fInfoPtr->mrfiPkt), sMyAddr, NET_ADDR_SIZE);
00210 
00211   return fInfoPtr;
00212 }

00285 /******************************************************************************
00286  * @fn          nwk_retrieveFrame
00287  *
00288  * @brief       Retrieve frame from Rx frame queue. Invoked by application-level
00289  *              code either app through SMPL_Receive() or IOCTL through raw Rx. This
00290  *              should run in a user thread, not an ISR thread.
00291  *
00292  * input parameters
00293  * @param    port    - port on which to get a frame
00294  *
00295  * output parameters
00296  * @param    msg     - pointer to where app payload should be copied. Buffer

00298  *
00299  * @param    len      - pointer to where payload length should be stored. Caller
00300  *                      can check for non-zero when polling the port. initialized
00301  *                      to 0 even if no frame is retrieved.

00303  *                      of the retrieved message.
00304  * @param    hopCount - if non-NULL, a pointer to where to copy the hop count
00305                         of the retrieved message.
00306  *
00307  * @return    SMPL_SUCCESS
00308  *            SMPL_NO_FRAME  - no frame found for specified destination
00309  *            SMPL_BAD_PARAM - no valid connection info for the Link ID
00310  *
00311  */
00312 smplStatus_t nwk_retrieveFrame(rcvContext_t *rcv, uint8_t *msg, uint8_t *len, addr_t *srcAddr, uint8_t *hopCount)
00313 {
00314   frameInfo_t *fPtr;
00315   uint8_t      done;
00316 
00317   do {
00318     /* look for a frame on requested port. */
00319     *len = 0;
00320     done = 1;
00321 
00322     fPtr = nwk_QfindOldest(INQ, rcv, USAGE_NORMAL);
00323     if (fPtr)
00324     {
00325       connInfo_t  *pCInfo = 0;
00326 
00327       if (RCV_APP_LID == rcv->type)
00328       {
00329         pCInfo = nwk_getConnInfo(rcv->t.lid);
00330         if (!pCInfo)
00331         {
00332           return SMPL_BAD_PARAM;
00333         }
00334 #if defined(SMPL_SECURE)
00335         /* decrypt here...we have all the context we need. */
00336         {
00337           uint32_t  ctr  = pCInfo->connRxCTR;
00338           uint32_t *pctr = &ctr;
00339           uint8_t   len  = MRFI_GET_PAYLOAD_LEN(&fPtr->mrfiPkt) - F_SEC_CTR_OS;
00340 
00341           if (pCInfo->thisLinkID == SMPL_LINKID_USER_UUD)
00342           {
00343             pctr = NULL;
00344           }
00345 #if defined(RX_POLLS)
00346           else if ((F_APP_PAYLOAD_OS - F_SEC_CTR_OS) == len)
00347           {
00348             /* This was an empty poll reply frame generated by the AP.
00349              * It uses the single-byte CTR value like network applications.
00350              * We do not want to use the application layer counter in this case.
00351              */
00352             pctr = NULL;
00353           }
00354 #endif
00355           if (nwk_getSecureFrame(&fPtr->mrfiPkt, len, pctr))
00356           {
00357             if (pctr)
00358             {
00359               /* Update connection's counter. */
00360               pCInfo->connRxCTR = ctr;
00361             }
00362           }
00363           else
00364           {
00365             /* Frame bogus. Check for another frame. */
00366             done = 0;
00367             continue;
00368           }
00369         }
00370 #endif  /* SMPL_SECURE */
00371       }
00372 
00373       /* it's on the requested port. */
00374       *len = MRFI_GET_PAYLOAD_LEN(&fPtr->mrfiPkt) - F_APP_PAYLOAD_OS;
00375       memcpy(msg, MRFI_P_PAYLOAD(&fPtr->mrfiPkt)+F_APP_PAYLOAD_OS, *len);
00376       /* save signal info */
00377       if (pCInfo)
00378       {
00379         /* Save Rx metrics... */
00380         pCInfo->sigInfo.rssi = fPtr->mrfiPkt.rxMetrics[MRFI_RX_METRICS_RSSI_OFS];
00381         pCInfo->sigInfo.lqi  = fPtr->mrfiPkt.rxMetrics[MRFI_RX_METRICS_CRC_LQI_OFS];
00382       }
00383       if (srcAddr)
00384       {
00385         /* copy source address if requested */
00386         memcpy(srcAddr, MRFI_P_SRC_ADDR(&fPtr->mrfiPkt), NET_ADDR_SIZE);
00387       }
00388       if (hopCount)
00389       {
00390         /* copy hop count if requested */
00391         *hopCount = GET_FROM_FRAME(MRFI_P_PAYLOAD(&fPtr->mrfiPkt), F_HOP_COUNT);
00392       }
00393       /* input frame no longer needed. free it. */
00394       nwk_QadjustOrder(INQ, fPtr->orderStamp);
00395 
00396       fPtr->fi_usage = FI_AVAILABLE;
00397       return SMPL_SUCCESS;
00398     }
00399   } while (!done);
00400 
00401   return SMPL_NO_FRAME;
00402 }

00405  * @fn          dispatchFrame
00406  *
00407  * @brief       Received frame looks OK so far. Dispatch to either NWK app by
00408  *              invoking the handler or the user's app by simply leaving the
00409  *              frame in the queue and letting the app poll the port.
00410  *
00411  * input parameters
00412  * @param   fiPtr    - frameInfo_t pointer to received frame
00413  *
00414  * output parameters
00415  *
00416  * @return   void
00417  */
00418 static void dispatchFrame(frameInfo_t *fiPtr)
00419 {
00420   uint8_t     port       = GET_FROM_FRAME(MRFI_P_PAYLOAD(&fiPtr->mrfiPkt), F_PORT_OS);
00421   uint8_t     nwkAppSize = sizeof(func)/sizeof(func[0]);
00422   fhStatus_t  rc;
00423   linkID_t    lid;
00424 #if defined(ACCESS_POINT)
00425   uint8_t loc;
00426 #endif
00427 #if !defined(END_DEVICE)
00428   uint8_t isForMe;
00429 #endif
00430 
00431   /* be sure it's not an echo... */
00432   if (!memcmp(MRFI_P_SRC_ADDR(&fiPtr->mrfiPkt), sMyAddr, NET_ADDR_SIZE))
00433   {
00434     fiPtr->fi_usage = FI_AVAILABLE;
00435     return;
00436   }
00437 
00438   /* Make sure encyrption bit conforms to our security support context. */
00439 #if defined(SMPL_SECURE)
00440   if (!(GET_FROM_FRAME(MRFI_P_PAYLOAD(&fiPtr->mrfiPkt), F_ENCRYPT_OS)))
00441   {
00442     /* Encyrption bit is not on when when it should be */
00443     fiPtr->fi_usage = FI_AVAILABLE;
00444     return;
00445   }
00446 #else
00447   if (GET_FROM_FRAME(MRFI_P_PAYLOAD(&fiPtr->mrfiPkt), F_ENCRYPT_OS))
00448   {
00449     /* Encyrption bit is on when when it should not be */
00450     fiPtr->fi_usage = FI_AVAILABLE;
00451     return;
00452   }
00453 #endif  /* SMPL_SECURE */
00454 
00455   /* If it's a network application port dispatch to service routine. Dispose
00456    * of frame depending on return code.
00457    */
00458   if (port && (port <= nwkAppSize))
00459   {
00460 #if defined(SMPL_SECURE)
00461     /* Non-connection-based frame. We can decode here if it was encrypted */
00462     if (!nwk_getSecureFrame(&fiPtr->mrfiPkt, MRFI_GET_PAYLOAD_LEN(&fiPtr->mrfiPkt) - F_SEC_CTR_OS, 0))
00463     {
00464       fiPtr->fi_usage = FI_AVAILABLE;
00465       return;
00466     }
00467 #endif
00468     rc = func[port-1](&fiPtr->mrfiPkt);
00469     if (FHS_KEEP == rc)
00470     {
00471       fiPtr->fi_usage = FI_INUSE_UNTIL_DEL;
00472     }
00473 #if !defined(END_DEVICE)
00474     else if (FHS_REPLAY == rc)
00475     {
00476       /* an AP or an RE could be relaying a NWK application frame... */
00477       nwk_replayFrame(fiPtr);
00478     }
00479 #endif
00480     else  /* rc == FHS_RELEASE (default...) */
00481     {
00482       fiPtr->fi_usage = FI_AVAILABLE;
00483     }
00484     return;
00485   }
00486   /* sanity check */
00487   else if ((port != SMPL_PORT_USER_BCAST) && ((port < PORT_BASE_NUMBER) || (port > SMPL_PORT_STATIC_MAX)))
00488   {
00489     /* bogus port. drop frame */
00490     fiPtr->fi_usage = FI_AVAILABLE;
00491     return;
00492   }
00493 
00494   /* At this point we know the target is a user app. If this is an end device
00495    * and we got this far save the frame and we're done. If we're an AP there
00496    * are 3 cases: it's for us, it's for s store-and-forward client, or we need
00497    * to replay the frame. If we're and RE and the frame didn't come from an RE
00498    * and it's not for us, replay the frame.
00499    */
00500 
00501 #if defined(END_DEVICE)
00502   /* If we're s polling end device we only accept application frames from
00503    * the AP. This prevents duplicate reception if we happen to be on when
00504    * a linked peer sends.
00505    */
00506 #if defined(RX_POLLS)
00507   if (F_TX_DEVICE_ED != GET_FROM_FRAME(MRFI_P_PAYLOAD(&fiPtr->mrfiPkt), F_TX_DEVICE))
00508   {
00509     if (nwk_isConnectionValid(&fiPtr->mrfiPkt, &lid))
00510     {
00511       fiPtr->fi_usage = FI_INUSE_UNTIL_DEL;
00512     }
00513     else
00514     {
00515       fiPtr->fi_usage = FI_AVAILABLE;
00516     }
00517   }
00518   else
00519   {
00520     fiPtr->fi_usage = FI_AVAILABLE;
00521   }
00522 #else
00523   /* it's destined for a user app. */
00524   if (nwk_isConnectionValid(&fiPtr->mrfiPkt, &lid))
00525   {
00526     fiPtr->fi_usage = FI_INUSE_UNTIL_DEL;
00527     if (spCallback && spCallback(lid))
00528     {
00529       fiPtr->fi_usage = FI_AVAILABLE;
00530       return;
00531     }
00532   }
00533   else
00534   {
00535     fiPtr->fi_usage = FI_AVAILABLE;
00536   }
00537 #endif  /* RX_POLLS */
00538 
00539 #else   /* END_DEVICE */
00540

00542    * handle this frame as if it were the target in case there is an application

00544    * replayed. It isn't enough just to test for the UUD port because it could be a
00545    * directed frame to another device. We must check explicitly for broadcast
00546    * destination address.
00547    */

00550   {
00551     /* The folllowing test will succeed for the UUD port regardless of the
00552      * source address.
00553      */
00554     if (nwk_isConnectionValid(&fiPtr->mrfiPkt, &lid))
00555     {
00556       /* If this is for the UUD port and we are here then the device is either
00557        * an AP or an RE. In either case it must replay the UUD port frame if the
00558        * frame is not "for me". But it also must handle it since it could have a
00559        * UUD-listening application. Do the reply first and let the subsequent code
00560        * correctly set the frame usage state. Note that the routine return can be
00561        * from this code block. If not it will drop through to the bottom without
00562        * doing a replay.
00563        */
00564       /* Do I need to replay it? */
00565       if (!isForMe)
00566       {
00567         /* must be a broadcast for the UUD port */
00568         nwk_replayFrame(fiPtr);
00569       }
00570       /* OK. Now I handle it... */
00571       fiPtr->fi_usage = FI_INUSE_UNTIL_DEL;
00572       if (spCallback && spCallback(lid))
00573       {
00574         fiPtr->fi_usage = FI_AVAILABLE;
00575         return;
00576       }
00577     }
00578     else
00579     {
00580       fiPtr->fi_usage = FI_AVAILABLE;
00581     }
00582   }

00586    */
