00097  * @return   void

00099 void nwk_globalsInit(void)
00100 {
00101 
00102   memset(&sAPAddress, 0x00, sizeof(addr_t));
00103 
00104   /* populate RAM address from ROM default if it hasn't laready been set
00105    * using the IOCTL interface.
00106    */
00107   if (!sRAMAddressIsSet)
00108   {
00109     memcpy(&sMyRAMAddress, &sMyROMAddress, sizeof(addr_t));
00110     sRAMAddressIsSet = 1;  /* RAM address is now valid */
00111   }
00112 
00113   return;

00116 /******************************************************************************
00117  * @fn          nwk_getMyAddress
00118  *
00119  * @brief       Return a pointer to my address. It comes either from ROM as
00120  *              set in the configuration file or it was set using the IOCTL
00121  *              interface -- probably from random bytes.
00122  *
00123  * input parameters
00124  *
00125  * output parameters
00126  *
00127  * @return   pointer to a constant address type object.
00128  */
00129 addr_t const *nwk_getMyAddress(void)
00130 {
00131   /* This call supports returning a valid pointer before either the
00132    * initialization or external setting of the address. But caller needs
00133    * to be careful -- if this routine is called immediately it will return
00134    * the ROM address. If the application then sets the address using the
00135    * IOCTL before doing the SMPL_Init() the original pointer is no longer
00136    * valid as it points to the wrong address.
00137    */
00138   return sRAMAddressIsSet ? &sMyRAMAddress : &sMyROMAddress;
00139 }
