00221 // @param       none
00222 // @return      none
00223 // *************************************************************************************************
00224 void do_altitude_measurement(u8 filter)
00225 {
00226     volatile u32 pressure;
00227 
00228     // If sensor is not ready, skip data read   
00229     if ((PS_INT_IN & PS_INT_PIN) == 0) return;
00230         
00231     // Get temperature (format is *10°K) from sensor
00232     sAlt.temperature = ps_get_temp();
00233 
00234     // Get pressure (format is 1Pa) from sensor
00235     pressure = ps_get_pa(); 
00236         
00237     // Store measured pressure value
00238     if (filter == FILTER_OFF) //sAlt.pressure == 0) 
00239     {
00240         sAlt.pressure = pressure;
00241     }
00242     else
00243     {
00244         // Filter current pressure
00245         pressure = (u32)((pressure * 0.2) + (sAlt.pressure * 0.8));
00246     
00247         // Store average pressure
00248         sAlt.pressure = pressure;
00249     }
00250     
00251     // Convert pressure (Pa) and temperature (°K) to altitude (m)
00252     sAlt.altitude = conv_pa_to_meter(sAlt.pressure, sAlt.temperature);
00253     
00254     // Calculate Ambient pressure, unit: milliBar = hectoPascal
00255     AmbientPressure = sAlt.pressure/100 + AmbientPressureOffset;
