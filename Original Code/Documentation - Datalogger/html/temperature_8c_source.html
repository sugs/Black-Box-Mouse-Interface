00092 // *************************************************************************************************
00093 // @fn          temperature_measurement
00094 // @brief       Init ADC12. Do single conversion of temperature sensor voltage. Turn off ADC12.
00095 // @param       none
00096 // @return      none
00097 // *************************************************************************************************
00098 void temperature_measurement(u8 filter)
00099 {
00100         u16 adc_result;
00101         volatile s32 temperature;
00102         
00103         // Convert internal temperature diode voltage 
00104         adc_result = adc12_single_conversion(REFVSEL_0, ADC12SHT0_8, ADC12INCH_10);
00105         
00106         // Convert ADC value to "xx.x °C"
00107         // Temperature in Celsius
00108     // ((A10/4096*1500mV) - 680mV)*(1/2.25mV) = (A10/4096*667) - 302
00109     // = (A10 - 1855) * (667 / 4096)
00110     temperature = (((s32)((s32)adc_result-1855))*667*10)/4096;
00111         
00112         // Add temperature offset
00113         temperature += sTemp.offset;    
00114         
00115         // Store measured temperature 
00116         if (filter == FILTER_ON)
00117         {
00118                 // Change temperature in 0.1° steps towards measured value
00119                 if (temperature > sTemp.degrees)                sTemp.degrees += 1;
00120                 else if (temperature < sTemp.degrees)   sTemp.degrees -= 1;
00121         }
00122         else
00123         {
00124                 // Override filter 
00125                 sTemp.degrees = (s16)temperature;
00126         }
00127 
00128         // New data is available --> do display update
00129         display.flag.update_temperature = 1;
